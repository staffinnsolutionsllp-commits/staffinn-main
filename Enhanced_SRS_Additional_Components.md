-- Platform Analytics Table
staffinn-platform-analytics
- analyticsId (String, Primary Key)
- date (String, Required)
- totalUsers (Number)
- activeUsers (Number)

- successfulPlacements (Number)
- revenue (Number)
- createdAt (String)

-- User Activity Logs Table
staffinn-user-activity-logs
- logId (String, Primary Key)
- userId (String, Foreign Key)
- action (String, Required)
- timestamp (String, Required)
- ipAddress (String)
- userAgent (String)
- sessionId (String)

-- Financial Transactions Table
staffinn-financial-transactions
- transactionId (String, Primary Key)
- userId (String, Foreign Key)
- amount (Number, Required)
- currency (String, Default: 'INR')
- transactionType (String, Required)
- paymentMethod (String)
- status (String, Required)
- createdAt (String, Required)

-- Notifications Table
staffinn-notifications
- notificationId (String, Primary Key)
- userId (String, Foreign Key)
- title (String, Required)
- message (String, Required)
- type (String, Required)
- isRead (Boolean, Default: false)
- createdAt (String, Required)

-- News Articles Table
staffinn-news-articles
- articleId (String, Primary Key)
- title (String, Required)
- content (String, Required)
- author (String, Required)
- category (String, Required)
- tags (Array)
- featuredImage (String)
- status (String, Default: 'draft')
- publishedAt (String)
- createdAt (String, Required)

-- System Configuration Table
staffinn-system-config
- configId (String, Primary Key)
- configKey (String, Required)
- configValue (String, Required)
- description (String)

- updatedAt (String, Required)
